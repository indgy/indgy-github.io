Effects

Effects supports media query modifiers (xs:, sm:, md:, lg:, xl:)

Synopsis

fx
{
 transition-property: background-color,border-color,color,fill,stroke,opacity,box-shadow,transform
}
fx-r
{
 transform-translate-x: 0;
 transform-translate-y: 0;
 transform-rotate: 0;
 transform-skew-x: 0;
 transform-skew-y: 0;
 transform-scale-x: 1;
 transform-scale-y: 1;
}
fx-[target] (background-color,border-color,color,fill,stroke, opacity, shadow, transform)
{
 transition-property: [target]
}
fx-mv-[dir]-[scale] (x, y), (0-19)
{
 transform: translate[dir]([scale]);
}
fx-sc-[scale] (0-200)
{
 transform: scale([scale], [scale]);
}
fx-rt-[angle] (0-360)
{
 transform: rotate([angle]deg);
}
-fx-rt-[angle] (0-360)
{
 transform: rotate(-[angle]deg);
}
fx-o-[position] (center, top, top right, right, bottom right, bottom, bottom left, left, top left)
{
 transform-origin: [position];
}
fx-dt-[duration] (0-1000)
{
 transition-duration: [duration]ms;
}
fx-dd-[duration] (0-1000)
{
 transition-delay: [duration]ms;
}
fx-e[ease] (linear, cubic-bezier(.4,0,1,1), cubic-bezier(0,0,.2,1), cubic-bezier(.4,0,.2,1))
{
 transition-timing-function: [ease];
}
